52 ! Calculate the coefficients b(i), c(i), and d(i), i=1,2,...,n
53 ! for cubic spline interpolation
54 ! s(x) = y(i) + b(i)*(x-x(i)) + c(i)*(x-x(i))**2 + d(i)*(x-x(i))**3
55 ! for x(i) <= x <= x(i+1)
56 ! Alex G: January 2010
57 !----------------------------------------------------------------------
58 ! input..
59 ! x = the arrays of data abscissas (in strictly increasing order)
60 ! y = the arrays of data ordinates
61 ! n = size of the arrays xi() and yi() (n>=2)
62 ! output..
63 ! b, c, d = arrays of spline coefficients
64 ! comments ...
65 ! spline.f90 program is based on fortran version of program spline.f
66 ! the accompanying function fspline can be used for interpolation
67 ! ==============================================================================
68 subroutine spline (x, y, b, c, d, n)
69  implicit none
70  integer n
71  real(dp) :: x(n), y(n), b(n), c(n), d(n)
72  integer i, j, gap
73  real :: h
74 
75  gap = n-1
76  ! check input
77  if ( n < 2 ) return
78  if ( n < 3 ) then
79  b(1) = (y(2)-y(1))/(x(2)-x(1)) ! linear interpolation
80  c(1) = 0.
81  d(1) = 0.
82  b(2) = b(1)
83  c(2) = 0.
84  d(2) = 0.
85  return
86  end if
87  !
88  ! step 1: preparation
89  !
90  d(1) = x(2) - x(1)
91  c(2) = (y(2) - y(1))/d(1)
92  do i = 2, gap
93  d(i) = x(i+1) - x(i)
94  b(i) = 2.0*(d(i-1) + d(i))
95  c(i+1) = (y(i+1) - y(i))/d(i)
96  c(i) = c(i+1) - c(i)
97  end do
98  !
99  ! step 2: end conditions
100  !
101  b(1) = -d(1)
102  b(n) = -d(n-1)
103  c(1) = 0.0
104  c(n) = 0.0
105  if(n /= 3) then
106  c(1) = c(3)/(x(4)-x(2)) - c(2)/(x(3)-x(1))
107  c(n) = c(n-1)/(x(n)-x(n-2)) - c(n-2)/(x(n-1)-x(n-3))
108  c(1) = c(1)*d(1)**2/(x(4)-x(1))
109  c(n) = -c(n)*d(n-1)**2/(x(n)-x(n-3))
110  end if
111  !
112  ! step 3: forward elimination
113  !
114  do i = 2, n
115  h = d(i-1)/b(i-1)
116  b(i) = b(i) - h*d(i-1)
117  c(i) = c(i) - h*c(i-1)
118  end do
119  !
120  ! step 4: back substitution
121  !
122  c(n) = c(n)/b(n)
123  do j = 1, gap
124  i = n-j
125  c(i) = (c(i) - d(i)*c(i+1))/b(i)
126  end do
127  !
128  ! step 5: compute spline coefficients
129  !
130  b(n) = (y(n) - y(gap))/d(gap) + d(gap)*(c(gap) + 2.0*c(n))
131  do i = 1, gap
132  b(i) = (y(i+1) - y(i))/d(i) - d(i)*(c(i+1) + 2.0*c(i))
133  d(i) = (c(i+1) - c(i))/d(i)
134  c(i) = 3.*c(i)
135  end do
136  c(n) = 3.0*c(n)
137  d(n) = d(n-1)
138 end subroutine spline

146 ! function ispline evaluates the cubic spline interpolation at point z
147 ! ispline = y(i)+b(i)*(u-x(i))+c(i)*(u-x(i))**2+d(i)*(u-x(i))**3
148 ! where x(i) <= u <= x(i+1)
149 !----------------------------------------------------------------------
150 ! input..
151 ! u = the abscissa at which the spline is to be evaluated
152 ! x, y = the arrays of given data points
153 ! b, c, d = arrays of spline coefficients computed by spline
154 ! n = the number of data points
155 ! output:
156 ! ispline = interpolated value at point u
157 !=======================================================================
158 function ispline(u, x, y, b, c, d, n)
159 implicit none
160 real ispline
161 integer n
162 real(dp):: u, x(n), y(n), b(n), c(n), d(n)
163 integer :: i, j, k
164 real :: dx
165 
166 ! if u is ouside the x() interval take a boundary value (left or right)
167 if(u <= x(1)) then
168  ispline = y(1)
169  return
170 end if
171 if(u >= x(n)) then
172  ispline = y(n)
173  return
174 end if
175 
176 !*
177 ! binary search for for i, such that x(i) <= u <= x(i+1)
178 !*
179 i = 1
180 j = n+1
181 do while (j > i+1)
182  k = (i+j)/2
183  if(u < x(k)) then
184  j=k
185  else
186  i=k
187  end if
188 end do
189 !*
190 ! evaluate spline interpolation
191 !*
192 dx = u - x(i)
193 ispline = y(i) + dx*(b(i) + dx*(c(i) + dx*d(i)))
194 end function ispline

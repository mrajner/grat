531 subroutine get_cmd_line_entry (dummy , cmd_line_entry , program_calling )
532  character(*) :: dummy
533  character(:), allocatable :: dummy2
534  type (cmd_line),intent(out) :: cmd_line_entry
535  character(1) :: separator=","
536  character(len=*) , intent(in) , optional :: program_calling
537  integer :: i , j , ii , jj
538 
539  cmd_line_entry%switch = dummy(1:2)
540  write(fileunit_tmp, form_61) , dummy
541  if (.not.if_switch_program(program_calling, cmd_line_entry%switch)) then
542  write ( fileunit_tmp , form_62 ) "this switch is IGNORED by program "//program_calling
543  return
544  endif
545 
546 
547  dummy=dummy(3:)
548 
549  cmd_line_entry%fields = count_separator(dummy) + 1
550  allocate(cmd_line_entry%field (cmd_line_entry%fields) )
551 
552  ! if ":" separator is present in command line allocate
553  ! additional array for fieldnames
554  if (count_separator(dummy, ":" ).ge.1) then
555  allocate(cmd_line_entry%fieldnames (cmd_line_entry%fields) )
556  endif
557  do i = 1 , cmd_line_entry%fields
558  j = index(dummy, separator)
559  cmd_line_entry%field(i) = dummy(1:j-1)
560  if (i.eq.cmd_line_entry%fields) cmd_line_entry%field(i)=dummy
561  dummy=dummy(j+1:)
562 
563  ! separate field and fieldnames
564  if ( index(cmd_line_entry%field(i),":").ne.0 ) then
565  dummy2 = trim(cmd_line_entry%field(i))//":"
566  allocate ( cmd_line_entry%fieldnames(i)%names(count_separator(dummy2,":") - 1 ))
567  do ii = 1, size(cmd_line_entry%fieldnames(i)%names)+1
568  jj = index(dummy2, ":")
569  if (ii.eq.1) then
570  cmd_line_entry%field(i) = dummy2(1:jj-1)
571  else
572  cmd_line_entry%fieldnames(i)%names(ii-1) = dummy2(1:jj-1)
573  endif
574  dummy2 = dummy2(jj+1:)
575  enddo
576  endif
577  enddo
578  call parse_option(cmd_line_entry , program_calling = program_calling)
579 end subroutine

103 ! ==============================================================================
104 !> This function computes the value of atmospheric gravity green functions
105 !! (AGGF) on the basis of spherical distance (psi)
106 !! \author Marcin Rajner
107 !! \date 2013.07.15
108 !! \warning psi in radians h in meter
109 !! t_zero is actually delta_t so if t_zero=10 (t_zero=288.15+10)
110 ! ==============================================================================
111 function aggf ( &
112  psi, &
113  zmin, &
114  zmax, &
115  dz, &
116  t_zero, &
117  h, &
118  first_derivative_h, &
119  first_derivative_z, &
120  fels_type, &
121  method, &
122  predefined, &
123  rough)
124 
125  use mod_constants, only: dp, pi, earth, gravity, atmosphere, R_air
126  use mod_utilities, only: d2r
127  use mod_atmosphere
128  use mod_normalization, only : green_normalization
129 
130  real(dp), intent(in) :: psi ! spherical distance from site [rad]
131  real(dp), intent(in),optional :: &
132  zmin, & ! minimum height, starting point [m] (default = 0)
133  zmax, & ! maximum height, ending point [m] (default = 60000)
134  dz, & ! integration step [m] (default = 0.1 -> 10 cm)
135  t_zero, & ! temperature at the surface [K] (default = 15°C i.e., 288.15=t0)
136  h ! station height [m] (default = 0)
137  logical, intent(in), optional :: &
138  first_derivative_h, first_derivative_z, predefined, rough
139  character (len=*), intent(in), optional :: fels_type, method
140  character (len=20) :: old_method
141  real(dp) :: aggf
142  real(dp) :: zmin_, zmax_, dz_, h_
143  real(dp) :: j_aux
144  real(dp) :: rho, l, deltat
145 
146  real(dp), dimension(:), allocatable, save :: heights, pressures
147  integer :: i
148 
149  zmin_ = 0.
150  zmax_ = 60000.
151  dz_ = 0.1
152  h_ = 0.
153 
154  aggf=0.
155 
156  if (present(zmin)) zmin_ = zmin
157  if (present(zmax)) zmax_ = zmax
158  if (present( dz)) dz_ = dz
159  if (present( h)) h_ = h
160  if (present(t_zero)) deltat=t_zero
161 
162  if(allocated(heights)) then
163  if ( &
164  ((zmin_ +dz_/2).ne.heights(1)) &
165  .or.abs((zmax_-dz_/2)-heights(size(heights))).gt.zmax_/1e6 &
166  .or.nint((zmax_-zmin_)/dz_).ne.size(heights) &
167  .or. (present(predefined)) &
168  .or. method.ne.old_method &
169  .or. present(t_zero) &
170  ) then
171  deallocate(heights)
172  deallocate(pressures)
173  endif
174  endif
175 
176  if (.not.allocated(heights)) then
177  allocate(heights(nint((zmax_-zmin_)/dz_)))
178  allocate(pressures(size(heights)))
179  do i = 1, size(heights)
180  heights(i) = zmin_ &
181  + dz_/2 &
182  + (i-1) * dz_
183  enddo
184 
185  if (present(rough).and.rough) then
186  ! do not use rough! it is only for testing
187  do i = 1, size(heights)
188  pressures(i) = standard_pressure( &
189  heights(i), &
190  method=method, &
191  dz=dz, &
192  use_standard_temperature=.true. &
193  )
194  enddo
195  else
196  pressures(1) = standard_pressure( &
197  heights(1), &
198  method = method, &
199  h_zero = zmin_, &
200  dz = dz, &
201  fels_type=fels_type, &
202  use_standard_temperature=.true., &
203  temperature = standard_temperature( &
204  zmin_, fels_type=fels_type)+deltat &
205  )
206  do i = 2, size(heights)
207  pressures(i) = standard_pressure( &
208  heights(i), &
209  p_zero = pressures(i-1), &
210  h_zero = heights(i-1), &
211  method = method, &
212  dz = dz, &
213  fels_type=fels_type, &
214  use_standard_temperature=.true., &
215  temperature = standard_temperature(heights(i-1), &
216  fels_type=fels_type)+deltat &
217  )
218  enddo
219  endif
220  endif
221  old_method=method
222 
223  do i = 1, size(heights)
224  l = ((earth%radius + heights(i))**2 + (earth%radius + h_)**2 &
225  - 2.*(earth%radius + h_)*(earth%radius+heights(i))*cos(psi))**(0.5)
226  rho = pressures(i)/ r_air / (deltat+standard_temperature(heights(i), fels_type=fels_type))
227  if (present(first_derivative_h) .and. first_derivative_h) then
228  ! first derivative (respective to station height)
229  ! micro Gal height / m
230  ! see equation 22, 23 in \cite Huang05
231  j_aux = ((earth%radius + heights(i) )**2)*(1.-3.*((cos(psi))**2)) -2.*(earth%radius + h_)**2 &
232  + 4.*(earth%radius+h_)*(earth%radius+heights(i))*cos(psi)
233  aggf = aggf + rho * ( j_aux / l**5 ) * dz_
234 
235  else if (present(first_derivative_z) .and. first_derivative_z) then
236  ! first derivative (respective to column height)
237  ! according to equation 26 in \cite Huang05
238  ! micro Gal / hPa / m
239  if (i.gt.1) exit
240  aggf = rho *( ((earth%radius + heights(i))*cos(psi)-(earth%radius + h_)) / (l**3))
241  else
242  ! GN microGal/hPa
243  aggf = aggf &
244  -rho*((earth%radius +heights(i))*cos(psi) - (earth%radius + h_)) / (l**3.) * dz_
245  endif
246  enddo
247  aggf = aggf/atmosphere%pressure%standard*gravity%constant*green_normalization("m", psi=psi)
248 end function

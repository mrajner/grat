107 !> This subroutine computes the value of atmospheric gravity green functions
108 !! (AGGF) on the basis of spherical distance (psi)
109 ! ==============================================================================
110 subroutine compute_aggf (psi , aggf_val , hmin , hmax , dh , if_normalization, &
111  t_zero , h , first_derivative_h , first_derivative_z , fels_type )
112  implicit none
113  real(dp), intent(in) :: psi !< spherical distance from site [degree]
114  real(dp), intent(in),optional :: hmin , & !< minimum height, starting point [km] (default=0)
115  hmax , & !< maximum height. eding point [km] (default=60)
116  dh , & !< integration step [km] (default=0.0001 -> 10 cm)
117  t_zero, & !< temperature at the surface [K] (default=288.15=t0)
118  h !< station height [km] (default=0)
119  logical, intent(in), optional :: if_normalization , first_derivative_h , first_derivative_z
120  character (len=*) , intent(in), optional :: fels_type
121  real(dp), intent(out) :: aggf_val
122  real(dp) :: r , z , psir , da , dz , rho , h_min , h_max , h_station , j_aux
123 
124  h_min = 0.
125  h_max = 60.
126  dz = 0.0001 !mrajner 2012-11-08 13:49
127  h_station = 0.
128 
129  if ( present(hmin) ) h_min = hmin
130  if ( present(hmax) ) h_max = hmax
131  if ( present( dh) ) dz = dh
132  if ( present( h) ) h_station = h
133 
134 
135  psir = psi * pi / 180.
136 
137  da = 2 * pi * r0**2 * ( 1 - cos(1. *pi/180.) )
138 
139 
140  aggf_val=0.
141  do z = h_min , h_max , dz
142 
143  r = ( ( r0 + z )**2 + (r0 + h_station)**2 &
144  - 2.*(r0 + h_station ) *(r0+z)*cos(psir) )**(0.5)
145  call standard_density( z , rho , t_zero = t_zero , fels_type = fels_type )
146 
147  !> first derivative (respective to station height)
148  !> micro Gal height / km
149  if ( present( first_derivative_h) .and. first_derivative_h ) then
150 
151  !! see equation 22, 23 in \cite Huang05
152  !J_aux = (( r0 + z )**2)*(1.-3.*((cos(psir))**2)) -2.*(r0 + h_station )**2 &
153  ! + 4.*(r0+h_station)*(r0+z)*cos(psir)
154  ! aggf_val = aggf_val - rho * ( J_aux / r**5 ) * dz
155 
156  !> direct derivative of equation 20 \cite Huang05
157  j_aux = (2.* (r0 ) - 2 * (r0 +z )*cos(psir)) / (2. * r)
158  j_aux = -r - 3 * j_aux * ((r0+z)*cos(psir) - r0)
159  aggf_val = aggf_val + rho * ( j_aux / r**4 ) * dz
160  else
161  !> first derivative (respective to column height)
162  !! according to equation 26 in \cite Huang05
163  !! micro Gal / hPa / km
164  if ( present( first_derivative_z) .and. first_derivative_z ) then
165  if (z.eq.h_min) then
166  aggf_val = aggf_val &
167  + rho*( ((r0 + z)*cos(psir) - ( r0 + h_station ) ) / ( r**3 ) )
168  endif
169  else
170  !> aggf GN
171  !! micro Gal / hPa
172  aggf_val = aggf_val &
173  + rho * ( ( (r0 + z ) * cos( psir ) - ( r0 + h_station ) ) / ( r**3 ) ) * dz
174  endif
175  endif
176  enddo
177 
178  aggf_val = -g * da * aggf_val * 1e8 * 1000
179 
180  !> if you put the optional parameter \c if_normalization=.false.
181  !! this block will be skipped
182  !! by default the normalization is applied according to \cite Merriam92
183  if ( (.not.present(if_normalization)) .or. (if_normalization)) then
184  aggf_val= psir * aggf_val * 1e5 / p0
185  endif
186 
187 end subroutine

212 ! =============================================================================
213 !> \brief Computes pressure [hPa] for specific height
214 !!
215 !! See \cite US1976 or \cite Huang05 for details.
216 !! Uses formulae 5 from \cite Huang05.
217 !! Simplified method if optional argument if_simplificated = .true.
218 ! =============================================================================
219 subroutine standard_pressure (height, pressure , &
220  p_zero , t_zero , h_zero, if_simplificated ,fels_type , inverted)
221  implicit none
222  real(dp) , intent(in) :: height
223  real(dp) , intent(in) , optional :: t_zero , p_zero , h_zero
224  character(len = 22) , optional :: fels_type
225  logical , intent(in) , optional :: if_simplificated
226  logical , intent(in) , optional :: inverted
227  real(dp), intent(out) :: pressure
228  real(dp) :: lambda , sfc_height , sfc_temperature , sfc_gravity , alpha , sfc_pressure
229 
230  sfc_temperature = t0
231  sfc_pressure = p0
232  sfc_height = 0.
233  sfc_gravity = g0
234 
235  if (present(h_zero)) then
236  sfc_height = h_zero
237  call standard_temperature(sfc_height , sfc_temperature )
238  call standard_temperature(sfc_height , sfc_temperature )
239  call standard_gravity(sfc_height , sfc_gravity )
240  endif
241 
242  if (present(p_zero)) sfc_pressure = p_zero
243  if (present(t_zero)) sfc_temperature = t_zero
244 
245  lambda = r_air * sfc_temperature / sfc_gravity
246 
247  if (present(if_simplificated) .and. if_simplificated ) then
248  ! use simplified formulae
249  alpha = -6.5
250  pressure = sfc_pressure &
251  * ( 1 + alpha / sfc_temperature * (height-sfc_height)) &
252  ** ( -sfc_gravity / (r_air * alpha / 1000. ) )
253  else
254  ! use precise formulae
255  pressure = sfc_pressure * exp( -1000. * (height -sfc_height) / lambda )
256  endif
257  if (present(inverted).and.inverted) then
258  pressure = sfc_pressure / ( exp( -1000. * (height-sfc_height) / lambda ) )
259  endif
260 end subroutine

325  temperature2, fels_type , tolerance)

334 
335  ! searching limits
336  temp(1)=t0-150
337  temp(3)=t0+ 50
338 
339  do

341

354  temp(1) = temp(2)

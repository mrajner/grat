126 subroutine nctime2date ( model )
127  type (file) :: model
128  real(dp):: mjd_start , mjd_
129  integer:: varid ,i , date (6)
130  character (len=66) :: dummy
131 
132 
133  call check( nf90_inq_varid(model%ncid, "time" , varid ) )
134  call check(nf90_get_att( model%ncid , varid , "units" , dummy) )
135 
136  allocate (model%date(size(model%time), 6))
137  write(log%unit , form_61) "Converting time: " , trim(dummy)
138  if (dummy.eq. "hours since 1-1-1 00:00:0.0") then
139  mjd_start = mjd([1,1,1,0,0,0])
140  do i = 1 , size(model%time)
141  mjd_= model%time(i) / 24 + mjd_start - 2
142  call invmjd(mjd_, date )
143  model%date(i,:)= date
144  enddo
145  else
146  write (log%unit , form_61 ) "unknown time begining"
147  endif

237 subroutine get_value( model, lat , lon , val ,level, method )
238 
239  type(file) , intent (in) :: model
240  real(sp) , intent (in) :: lat,lon
241  real(sp) , intent(out) :: val
242  integer , optional , intent(in) :: method
243  integer , optional , intent(in) :: level
244  integer :: i , ilevel =1
245  integer :: ilon, ilat , ilon2 , ilat2
246  real(sp), dimension(4,3) :: array_aux
247 
248  if (present(level)) ilevel=level
249  ! check if inside model range
250  if ( lat.lt.min(model%latrange(1), model%latrange(2)) &
251  .or.lat.gt.max(model%latrange(1), model%latrange(2)) &
252  .or.lon.lt.min(model%latrange(1), model%latrange(2)) &
253  .or. (lon.gt.max(model%latrange(1), model%latrange(2)) &
254  .and. lon.gt.max(model%latrange(1), model%latrange(2))) &
255  ) then
256  val = sqrt(-1.)
257  endif
258  if (model%if_constant_value) then
259  val = model%constant_value
260  return
261  end if
262 
263  ilat = minloc(abs(model%lat-lat),1)
264  ilon = minloc(abs(model%lon-lon),1)
265 
266  if (present(method) .and. method .eq. 2 ) then
267  ilon2 = minloc(abs(model%lon-lon),1, model%lon/=model%lon(ilon))
268  ilat2 = minloc(abs(model%lat-lat),1, model%lat/=model%lat(ilat))
269 
270  if (lon.gt.model%lon(ilon2).and. lon.gt.model%lon(ilon)) then
271  else
272  array_aux(1, :) = [ model%lon(ilon) , model%lat(ilat) , model%data(ilon , ilat , ilevel) ]
273  array_aux(2, :) = [ model%lon(ilon) , model%lat(ilat2), model%data(ilon , ilat2, ilevel) ]
274  array_aux(3, :) = [ model%lon(ilon2), model%lat(ilat) , model%data(ilon2, ilat , ilevel) ]
275  array_aux(4, :) = [ model%lon(ilon2), model%lat(ilat2), model%data(ilon2, ilat2, ilevel) ]
276  if (moreverbose%if.and.moreverbose%names(1).eq."b") then
277  write(moreverbose%unit , '(3f15.4)') , (array_aux(i,:), i = 1 ,4)
278  write(moreverbose%unit , '(">")')
279  endif
280  val = bilinear( lon , lat , array_aux )
281  return
282  endif
283  endif
284 
285  ! if the last element is the neares then check if the firt is not nearer
286  ! i.e. search in 0-357.5E for 359E
287  if (ilon .eq. size (model%lon) ) then
288  if (abs(model%lon(ilon)-lon).gt.abs(model%lon(1)+360.-lon)) ilon = 1
289  endif
290  if (moreverbose%if.and.moreverbose%names(1).eq."n") then
291  write(moreverbose%unit , '(3f15.4)') , &
292  model%lon(ilon) , model%lat(ilat) , model%data(ilon,ilat,ilevel)
293  write(moreverbose%unit , '(">")')
294  endif
295  val = model%data (ilon , ilat, ilevel)
296 end subroutine

298 real function bilinear (x , y , aux )
299  real x , y , aux(4,3)
300  real(sp) :: a , b , c
301  a = ( x - aux(1,1) ) /(aux(4,1)-aux(1,1))
302  b = a * (aux(3,3) - aux(1,3)) + aux(1,3)
303  c = a * (aux(4,3) - aux(2,3)) + aux(2,3)
304  bilinear = (y-aux(1,2))/(aux(4,2) -aux(1,2)) * (c-b) + b
305 end function

36 ! =============================================================================
37 !> Parse date given as 20110503020103 to yy mm dd hh mm ss and mjd
38 !!
39 !! \warning decimal seconds are not allowed
40 ! =============================================================================
41 subroutine parse_date(cmd_line_entry)
42  use mod_cmdline
43  use mod_mjd, only: mjd, invmjd
44  use mod_utilities, only: is_numeric
45  use mod_data, only: model
46 
47  integer, dimension(6) :: start, stop, swap
48  real (dp) :: step
49  integer :: i_, i, start_index, i_aux
50  character(1) :: interval_unit
51  type(cmd_line_arg) :: cmd_line_entry
52  logical :: success
53 
54  if (allocated(date)) then
55  call print_warning("repeated")
56  return
57  endif
58 
59  do i_ = 1, size(cmd_line_entry%field)
60 
61  if (trim(cmd_line_entry%field(i_)%full).eq."") then
62  call print_warning("bad date " //trim(cmd_line_entry%field(i_)%full))
63  cycle
64  endif
65 
66  do i_aux=1, min(size(cmd_line_entry%field(i_)%subfield),2)
67  if(index(cmd_line_entry%field(i_)%subfield(i_aux)%name,'-').gt.1) then
68  call strip_hyphen_date_iso(cmd_line_entry%field(i_)%subfield(i_aux)%name)
69  endif
70  enddo
71 
72  if (any([(cmd_line_entry%field(i_)%subfield(i_aux)%name.ne."m" &
73  .and..not.is_numeric(cmd_line_entry%field(i_)%subfield(i_aux)%name), &
74  i_aux=1, size(cmd_line_entry%field(i_)%subfield))])) then
75  call print_warning( &
76  "date not numeric "// trim(cmd_line_entry%field(i_)%full), &
77  error=.true. &
78  )
79  cycle
80  endif
81 
82  if (any( [( &
83  is_numeric(cmd_line_entry%field(i_)%subfield(i_aux)%name) &
84  .and.index(cmd_line_entry%field(i_)%subfield(i_aux)%name,".").ne.0 , &
85  i_aux=1, size(cmd_line_entry%field(i_)%subfield))])) then
86  call print_warning("decimal date not supported "// trim(cmd_line_entry%field(i_)%full))
87  cycle
88  endif
89 
90  interval_unit = "h"
91 
92  if (.not.log%sparse) &
93  write(log%unit, form%i2) trim(cmd_line_entry%field(i_)%full)
94 
95  if (cmd_line_entry%field(i_)%subfield(1)%name.eq."m") then
96  if (size(model(1)%date).eq.0) then
97  call print_warning("NO dates in first model. -Dm is forbidden (or -D before -F)", error=.true.)
98  else
99  start = model(1)%date(lbound(model(1)%date, 1), 1:6)
100  endif
101  else
102  call string2date(cmd_line_entry%field(i_)%subfield(1)%name, start, success=success)
103  if (.not.success) cycle
104  endif
105 
106  stop = start
107 
108  ! tilde
109  if (cmd_line_entry%field(i_)%subfield(1)%dataname=="~") then
110  swap = model(1)%date(1, :)
111  do i=2, size(model(1)%date(:, :), 1)
112  if (abs(mjd(model(1)%date(i, :))-mjd(start)).lt.abs(mjd(swap)-mjd(start))) then
113  swap = model(1)%date(i, :)
114  endif
115  enddo
116  start = swap
117  endif
118 
119  if (size(cmd_line_entry%field(i_)%subfield).ge.2 &
120  .and. cmd_line_entry%field(i_)%subfield(2)%name.ne."" &
121  ) then
122  if (cmd_line_entry%field(i_)%subfield(2)%name.eq."m") then
123  stop = model(1)%date(ubound(model(1)%date, 1), 1:6)
124  else if (cmd_line_entry%field(i_)%subfield(2)%dataname.ne."") then
125  read (cmd_line_entry%field(i_)%subfield(2)%name,*) stop(1)
126  select case (cmd_line_entry%field(i_)%subfield(2)%dataname)
127  case('Y')
128  stop(1)=start(1)+stop(1)
129  stop(2:)=start(2:)
130  case('M')
131  stop(2)=start(2)+stop(1)
132  stop(1)=start(1)
133  stop(3:)=start(3:)
134  if (stop(2).gt.12) then
135  stop(1) = stop(1)+int(stop(2)/12)
136  stop(2) = modulo(stop(2), 12)
137  else if (stop(2).lt.1) then
138  stop(1)=stop(1)-int(-stop(2)/12+1)
139  stop(2)=stop(2)+12*(1+int(-stop(2)/12))
140  endif
141  case('D')
142  call invmjd(mjd(start)+stop(1), stop)
143  case('H')
144  call invmjd(mjd(start)+stop(1)/24., stop)
145  case('')
146  case default
147  call print_warning("unit not valid", error=.true.)
148  cycle
149  endselect
150  else
151  call string2date(cmd_line_entry%field(i_)%subfield(2)%name, stop)
152  endif
153  else
154  stop = start
155  endif
156 
157  if (size(cmd_line_entry%field(i_)%subfield).ge.3)then
158  read (cmd_line_entry%field(i_)%subfield(3)%name, *) step
159  select case (cmd_line_entry%field(i_)%subfield(3)%dataname)
160  case("M", "D", "Y", "H")
161  read (cmd_line_entry%field(i_)%subfield(3)%dataname, * ) interval_unit
162  case default
163  call print_warning("interval unit not valid", error=.true.)
164  cycle
165  endselect
166  else
167  step=6
168  endif
169 
170  if (.not.log%sparse) then
171  write (log%unit, '('//form%t3//', a, x, i4, 5(1x, i2.2))') "start date:", start
172 
173  if (mjd(start).ne.mjd(stop)) then
174  write (log%unit, '('//form%t3//', a, x, i4, 5(1x, i2.2))') "stop date:", stop
175  write (log%unit, "(" // form%t3// "a, f5.1, a)") "interval:", step, interval_unit
176  endif
177  endif
178 
179  ! allow that stop is previous than start and list in reverse order
180  ! chage the sign of step in dates if necessery
181  if(mjd(stop).lt.mjd(start).and. step.gt.0) step = -step
182  ! or if step is negative
183  if(mjd(stop).gt.mjd(start).and. step.lt.0) then
184  swap=start
185  start=stop
186  stop=swap
187  endif
188 
189  if (interval_unit.eq."M".or.interval_unit.eq."Y") then
190  if (interval_unit.eq."Y") then
191  step=step*12
192  interval_unit="M"
193  endif
194  if (interval_unit.eq."M") then
195  call more_dates &
196  (int((12*(stop(1) - start(1))+stop(2)-start(2))/(step))+1, start_index)
197  date(start_index)%date=start
198  date(start_index)%mjd=mjd(date(start_index)%date)
199  do i= start_index+1, size(date)
200  date(i)%date=date(i-1)%date
201  date(i)%date(2)=date(i-1)%date(2)+step
202  if (date(i)%date(2).gt.12) then
203  date(i)%date(1) =date(i)%date(1)+int(date(i)%date(2)/12)
204  date(i)%date(2) =modulo(date(i)%date(2), 12)
205  else if (date(i)%date(2).lt.1) then
206  date(i)%date(1) =date(i)%date(1)-int(-date(i)%date(2)/12+1)
207  date(i)%date(2) =date(i)%date(2)+12*(1+int(-date(i)%date(2)/12))
208  endif
209  date(i)%mjd=mjd(date(i)%date)
210  call invmjd(date(i)%mjd, date(i)%date)
211  enddo
212  endif
213  else
214  if (cmd_line_entry%field(i_)%subfield(1)%name=="m" &
215  .and. cmd_line_entry%field(i_)%subfield(2)%name=="m" &
216  .and. ( &
217  size(cmd_line_entry%field(i_)%subfield).lt.3 .or. &
218  cmd_line_entry%field(i_)%subfield(3)%dataname=="" &
219  ) &
220  ) then
221  if (size(cmd_line_entry%field(i_)%subfield).lt.3) step=1
222  if (step.gt.size(model(1)%time)) step=size(model(1)%time)
223  call more_dates(ceiling(size(model(1)%time)/step), start_index)
224  i_aux=0
225  do i = 1, size(model(1)%time), int(step)
226  i_aux=i_aux+1
227  date(i_aux)%date = model(1)%date(i, :)
228  date(i_aux)%mjd =mjd(date(i_aux)%date)
229  enddo
230  else
231  if (interval_unit.eq."D") step = 24. * step
232  if (interval_unit.eq."m") step = step/60.
233  if (interval_unit.eq."s") step = step/60./60.
234 
235  call more_dates(int((mjd(stop)-mjd(start)) / step * 24. + 1 ), start_index)
236  do i = start_index, size(date)
237  date(i)%mjd = mjd(start) + (i-start_index)*step/24.
238  call invmjd(date(i)%mjd, date(i)%date)
239  enddo
240  endif
241  endif
242  enddo
243  if (.not.log%sparse) &
244  write (log%unit, form%i3) "dates total:", size(date)
245 
246 end subroutine

268 ! =============================================================================
269 !> Convert dates given as string to integer (6 elements)
270 !!
271 !! 20110612060302 --> [2011, 6, 12, 6, 3, 2 ]
272 !! you can omit
273 !! \warning decimal seconds are not allowed
274 ! =============================================================================
275 subroutine string2date (string, date, success)
276  use mod_utilities, only: is_numeric
277 
278  character (*), intent(in) :: string
279  integer, dimension(6), intent(out):: date
280  integer :: start_char, end_char, j
281  logical, optional :: success
282 
283  if (present(success)) success=.true.
284 
285  ! this allow to specify !st Jan of year simple as -Dyyyy
286  date = [2000, 1, 1, 0, 0, 0]
287 
288  start_char = 1
289  do j = 1, 6
290  if (j.eq.1) then
291  end_char=min(len(string), start_char+3)
292  else
293  end_char=min(len(string), start_char+1)
294  endif
295  if (is_numeric(string(start_char : end_char) )) then
296  read(string(start_char : end_char), *) date(j)
297  else
298  call print_warning("bad date " // string)
299  if (present(success)) success=.false.
300  return
301  endif
302  start_char=end_char+1
303  if (end_char.eq.len(trim(string))) exit
304  enddo
305 end subroutine

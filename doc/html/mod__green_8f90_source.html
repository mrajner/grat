66 !> This soubroutine gives the latitude and longitude of the point at the
67 !! specified distance and azimuth from site latitude and longitude.
68 !!
69 !! Adopted from \cite Agnew97
70 !! all parameters in decimal degree
71 ! =============================================================================
72 subroutine spher_trig ( latin , lonin , distance , azimuth , latout , lonout)
73  real(dp) , intent(in) :: distance
74  real(sp) , intent(in) :: latin , lonin , azimuth
75  real(dp) , intent(out) :: latout, lonout
76  real(dp):: sg, cg , saz ,caz , st ,ct , cd ,sd , cb , sb
77 
78  ct = cos(d2r(90.-dble(latin)))
79  st = sin(d2r(90.-dble(latin)))
80  cd = cos(d2r(distance))
81  sd = sin(d2r(distance))
82  saz = sin(d2r(dble(azimuth)))
83  caz = cos(d2r(dble(azimuth)))
84  cb = cd*ct + sd*st*caz
85  !todo !if(abs(cb).gt.1) cb = cb/abs(cb)
86  sb = sqrt(1.-cb**2)
87  latout = 90 - r2d(acos(cb))
88  lonout = lonin + r2d(atan2(sd*saz/sb,(st*cd - sd*ct*caz)/sb))
89 end subroutine

93 subroutine convolve (site , green , results, denserdist , denseraz )
94  type(site_data) , intent(in) :: site
95  type(green_functions), allocatable , dimension(:) :: green
96  integer , intent (in) :: denserdist , denseraz
97  real(sp) :: latin , lonin
98  integer :: ndenser , igreen , iazimuth , nazimuth
99  real(sp) :: azimuth
100  real(dp) :: lat , lon , area
101  real(sp) :: val(4) , ref_p
102  integer :: i , iok(2) , npoints
103  real(dp) :: normalize
104  type (result) ,intent(out) :: results
105 
106  if (.not.allocated(green_common)) then
107  call green_unification(green , green_common , denser = denserdist-1)
108  endif
109 
110  npoints=0
111  do igreen = 1 ,size(green_common(:,1))
112  nazimuth = max(int(360*sin(d2r(green_common(igreen,1)))),100) * denseraz
113  do iazimuth = 1 , nazimuth
114  npoints = npoints + 1
115  azimuth = (iazimuth - 1) * 360./nazimuth
116 
117  ! get lat and lon of point
118  call spher_trig( site%lat , site%lon , green_common(igreen,1) , azimuth , lat , lon)
119 
120  ! get values of model
121 
122  do i = 1 , size(model)
123  if(model(i)%if) then
124  call get_value(model(i) , real(lat) , real(lon) , val(i) , level=1, method =model(i)%interpolation)
125  else
126  val(i) = 0.
127  endif
128  enddo
129 
130  if (refpres%if) then
131  call get_value(refpres , real(lat) , real(lon) , ref_p , method =1)
132  else
133  ref_p=0.
134  endif
135 
136  ! get polygons
137  do i = 1 , 2
138  if (polygons(i)%if) then
139  call chkgon( real(lon), real(lat) , polygons(i) , iok(i) )
140  else
141  iok(i)=1
142  endif
143  end do
144 
145  ! calculate area using spherical formulae
146  if (val(1).ne.0) then
147  call spher_area(green_common(igreen,1) , green_common(igreen,2), 360./nazimuth , area)
148 
149  ! force topography to zero over oceans
150  if (val(4).eq.0.and.val(3).lt.0) val(3) = 0.
151 
152  ! normalization according to Merriam (1992)
153  normalize= 1. / ( 2. * pi * ( 1. - cos( d2r(dble(1.)) ) ) * d2r(green_common(igreen,1)) *1.e5 )
154 
155  ! elastic part
156  ! if the cell is not over sea and inverted barometer assumption was not set
157  ! and is not excluded by polygon
158  if ((.not.((val(4).eq.0.and.inverted_barometer).or. iok(2).eq.0)).or.size(model).lt.4) then
159  results%e = results%e + (val(1) / 100. -ref_p) * green_common(igreen,7) * area * normalize
160  endif
161  ! print*, results%e , inverted_barometer , .not.((val(4).eq.0.and.inverted_barometer).or. iok(2).eq.0) ,val(4)
162  ! stop
163 
164  ! newtonian part
165  if(.not. iok(1).eq.0) then
166  results%n = results%n + (val(1)/ 100.-ref_p) * green_common(igreen,3) * area * normalize
167 
168  if (model(2)%if.and.size(model).ge.2) then
169  results%dt = results%dt + (val(1)/ 100.-ref_p) * &
170  (green_common(igreen,4)*(val(2)-t0) ) * area * normalize
171  endif
172 
173  results%dh = results%dh + (val(1)/ 100.-ref_p) * &
174  (green_common(igreen,5)*(site%height/1000.) ) * area * normalize
175 
176  results%dz = results%dz + (val(1)/ 100.-ref_p) * &
177  (green_common(igreen,6)*(val(3)/1000.) ) * area * normalize
178  endif
179  endif
180  if (moreverbose%if.and. moreverbose%names(1).eq."g") then
181  call convolve_moreverbose(site%lat,site%lon , azimuth , 360./ nazimuth , green_common(igreen,1), green_common(igreen,1))
182  write (moreverbose%unit, '(">")')
183  endif
184  enddo
185  enddo
186  if (moreverbose%if.and. moreverbose%names(1).eq."i") then
187  write (moreverbose%unit, '(a,x,g0)') "Points used in convolution" ,npoints
188  endif
189 end subroutine

214 subroutine spher_trig_inverse (lat1, lon1, lat2 , lon2 , distance , azimuth, haversine)
215  real(dp) , intent (in) :: lat1 , lon1 , lat2 , lon2
216  real(dp) , intent (out) :: distance , azimuth
217  real(dp) :: dlat,dlon
218  logical, intent(in) , optional :: haversine
219 
220  dlon = lon2 -lon1
221  dlat = lat2 -lat1
222 ! if (dlon > pi ) dlon =dlon -pi
223 
224  if (present(haversine)) then
225  distance = 2 * asin( sqrt( (sin(dlat/2))**2 + cos(lat1) * cos(lat2) * (sin(dlon/2))**2 ) )
226 
227  else
228  distance = acos( sin(lat1) * sin(lat2) + cos(lat1) * cos(lat2) * cos(dlon))
229  endif
230 
231  azimuth = atan2( (sin(dlon)*cos(lat2)/sin(distance)) , ((sin(lat2)*cos(lat1) - cos(lat2)*sin(lat1)*cos(dlon))/sin(distance)) )
232  if (azimuth.lt.0) azimuth = azimuth + 2 * pi
233 
234 end subroutine

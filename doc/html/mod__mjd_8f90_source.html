9 ! ==============================================================================
10 !> Compute date from given Julian Day
11 !!
12 !! This subroutine computes date (as an six elements integer array) from
13 !! Modified Julian Day
14 !! \date 2013-03-04
15 ! ==============================================================================
16 subroutine invmjd (mjd, date)
17  real(dp), intent (in) :: mjd
18  integer, intent (out):: date (6)
19  integer :: t1, t4, h, t2, t3, ih1, ih2
20  real(dp) :: dayfrac
21 
22  date =0
23 
24  t1 = 1+ int(mjd) + 2400000
25  t4 = mjd - int(mjd);
26  h = int((t1 - 1867216.25)/36524.25)
27  t2 = t1 + 1 + h - int(h/4)
28  t3 = t2 - 1720995
29  ih1 = int((t3 -122.1)/365.25)
30  t1 = int(365.25 * ih1)
31  ih2 = int((t3 - t1)/30.6001);
32  date(3) = (t3 - t1 - int(30.6001 * ih2)) + t4
33  date(2) = ih2 - 1
34 
35  if (ih2 .gt. 13) date(2) = ih2 - 13
36  date(1) = ih1
37  if (date(2).le. 2) date(1) = date(1) + 1
38 
39  dayfrac = mjd - int(mjd) + 1./ (60*60*1000)
40  date(4) = int(dayfrac * 24.)
41  date(5) = (dayfrac - date(4) / 24.) * 60 * 24
42  date(6) = (dayfrac - date(4) / 24. - date(5)/(24.*60.) ) * 60 * 24 *60
43 
44  if (date(6) .eq. 60 ) then
45  date(6) = 0
46  date(5) = date(5) + 1
47  endif
48 end subroutine

89

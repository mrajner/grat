450 subroutine check_arguments (program_calling)
451  use mod_date, only: date
452  use mod_data, only: model, parse_level
453  use mod_cmdline, only: cmd_line, method, quiet, ind, transfer_sp, &
454  inverted_barometer
455  use mod_site, only: gather_site_model_info
456  use mod_green, only: parse_green
457  character(len=*), intent(in) :: program_calling
458  integer :: i
459 
460  if (program_calling.eq."grat") then
461 
462  ! assume method using for computation if not given explicitly
463  if (.not.any(cmd_line%switch.eq.'-M')) then
464  if (any(cmd_line%switch.eq.'-G')) then
465  method(2)=.true.
466  if (.not.quiet) call print_warning("method", more= "assuming 2D")
467  else
468  method(1)=.true.
469  if (.not.quiet) call print_warning("method", more= "assuming 1D")
470  endif
471  endif
472 
473  if (method(2) .and. .not.any(cmd_line%switch.eq.'-G')) then
474  call print_warning("green_missing", error=.true.)
475  endif
476 
477  if (method(3)) then
478  if(.not.any(cmd_line%switch.eq.'-G')) then
479  call print_warning("no method 2D, so 3D result will be shifted")
480  call parse_green()
481  endif
482 
483  if (.not.any(cmd_line%switch.eq.'-J')) then
484  call parse_level()
485  endif
486  endif
487 
488 
489  if (((method(2).or.method(3)) &
490  .and. inverted_barometer) &
491  .and. (ind%model%ls.eq.0 &
492  .or.(.not.model(ind%model%ls)%if &
493  .and..not.model(ind%model%ls)%if_constant_value) &
494  ) &
495  .and. ind%green%ge.ne.0) then
496  call print_warning( &
497  "inverted barometer, but no landsea mask", &
498  error=.true. &
499  )
500  endif
501 
502  endif
503 
504  do i=1, size(model)
505  if (model(i)%autoload) then
506  if (.not. allocated(date)) then
507  call print_warning("alias_without_date", error=.true.)
508  endif
509  endif
510  enddo
511 
512  call gather_site_model_info()
513 
514  if (ind%model%hp.ne.0.and. .not.transfer_sp%if) then
515  call print_warning("maybe use -U")
516  endif
517 
518  if (transfer_sp%if .and. ind%model%hp.eq.0) then
519  call print_warning("-U but no @HP found")
520  endif
521 
522  ! just to avoid error when no levels are found
523  if (method(3)) then
524  if (any([ind%model%gp].eq.0)) then
525  call print_warning("@GP is missing", error=.true.)
526  endif
527  endif
528 
529  if (.not.any(cmd_line%switch.eq.'-S')) then
530  call print_warning("-S not given")
531  endif
532 end subroutine

536 !! is not
537 ! =============================================================================
538 logical function if_accepted_switch (switch, accepted_switches)
539  character(len= *), intent (in) :: switch
540  character(len= *), intent (in), optional :: accepted_switches
541  integer :: i
542 
543  if (.not.present(accepted_switches)) then
544  if_accepted_switch=.true.
545  return
546  endif
547 
548  ! default
549  if_accepted_switch=.false.
550 
551  ! loop trough accepted switches
552  do i =1, len(accepted_switches)
553  if (switch(2:2).eq.accepted_switches(i:i)) then
554  if_accepted_switch=.true.
555  return
556  endif
557  enddo
558 end function

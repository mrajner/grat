23 subroutine read_polygon (polygon)
24  use mod_cmdline
25  use mod_utilities, only: skip_header
26  type(polygon_info):: polygon
27  integer :: i , j , number_of_polygons , nvertex
28  character(80) :: dummy
29  character (1) :: pm
30 
31 
32  if (polygon%if) then
33  ! polygon file
34  open (newunit = polygon%unit , action="read", file=polygon%name )
35 
36  ! first get the number of polygons
37  call skip_header(polygon%unit)
38  read (polygon%unit , * ) number_of_polygons
39  allocate (polygon%polygons(number_of_polygons))
40 
41  ! loop over all polygons in file
42  do i=1, number_of_polygons
43  call skip_header(polygon%unit)
44  read (polygon%unit, * ) nvertex
45  allocate (polygon%polygons(i)%coords(nvertex, 2 ))
46  call skip_header(polygon%unit)
47  read (polygon%unit, * ) pm
48  if (pm.eq."+") polygon%polygons(i)%use=.true.
49  if (pm.eq."-") polygon%polygons(i)%use=.false.
50  ! override file +|- with global given with command line
51  if (polygon%pm.eq."+") polygon%polygons(i)%use=.true.
52  if (polygon%pm.eq."-") polygon%polygons(i)%use=.false.
53  do j = 1 , nvertex
54  call skip_header(polygon%unit)
55  ! lon lat , checks while reading
56  read (polygon%unit, * ) polygon%polygons(i)%coords(j,1:2)
57  if ( polygon%polygons(i)%coords(j,1).lt.-180. &
58  .or.polygon%polygons(i)%coords(j,1).gt.360. &
59  .or.polygon%polygons(i)%coords(j,2).lt.-90. &
60  .or.polygon%polygons(i)%coords(j,2).gt. 90. ) then
61  write (error_unit , form_63) "Somethings wrong with coords in polygon file"
62  call exit
63  elseif( polygon%polygons(i)%coords(j,1).lt.0. ) then
64  polygon%polygons(i)%coords(j,1) = polygon%polygons(i)%coords(j,1) + 360.
65  endif
66  enddo
67  enddo
68  close (polygon%unit)
69 
70  ! print summary to log file
71  write (log%unit, form_separator)
72  write (log%unit, form_60) "Summary of polygon file"
73  write (log%unit, form_61) "name:", trim(polygon%name)
74  write (log%unit, form_61) "number of polygons:" , size (polygon%polygons)
75  do i = 1 , size (polygon%polygons)
76  if (polygon%pm.eq."+".or.polygon%pm.eq."-") write (log%unit, form_62) &
77  "Usage overwritten with command line option", polygon%pm
78  write (log%unit, form_62) "use [true/false]:" , &
79  polygon%polygons(i)%use
80  write (log%unit, form_62) "number of coords:" , &
81  size (polygon%polygons(i)%coords(:,1))
82  enddo
83  endif

88 ! ==============================================================================
89 !> Check if point is in closed polygon
90 !!
91 !! From spotl \cite Agnew97
92 !! adopted to grat and Fortran90 syntax
93 !! From original description
94 !! returns iok=0 if
95 !! 1. there is any polygon (of all those read in) in which the
96 !! coordinate should not fall, and it does
97 !! or
98 !! 2. the coordinate should fall in at least one polygon
99 !! (of those read in) and it does not
100 !! otherwise returns iok=1
101 !! \author D.C. Agnew \cite Agnew96
102 !! \author adopted by Marcin Rajner
103 !! \date 2013-03-04
104 ! ==============================================================================
105 subroutine chkgon (rlong , rlat , polygon , iok)
106  use mod_constants, only: dp , dp
107  use mod_cmdline
108  real(dp),intent (in) :: rlong, rlat
109  integer :: i ,ii , ianyok
110  integer , intent (out) :: iok
111  real(dp) :: rlong2
112  type( polygon_info ) , intent (in) :: polygon
113 
114  ! Check first if we need to use this soubroutine
115  if (size(polygon%polygons).eq.0) then
116  iok=0
117  return
118  endif
119 
120  if(rlong.gt.180) rlong2 = rlong - 360.
121  ! loop over polygons
122  do i=1,size(polygon%polygons)
123  ! loop twice for elastic and newtonian
124  ! polygon is one we should not be in
125  if(.not.polygon%polygons(i)%use) then
126  if ( if_inpoly(rlong ,rlat,polygon%polygons(i)%coords).ne.0 &
127  .or.if_inpoly(rlong2 ,rlat,polygon%polygons(i)%coords).ne.0 ) then
128  iok=0
129  return
130  endif
131  endif
132  enddo
133  ianyok=0
134  ! polygon is one we should be in; test to see if we are, and if so set
135  ! iok to 1 and return
136  do i=1,size(polygon%polygons)
137  if(polygon%polygons(i)%use) then
138  ianyok = ianyok+1
139  if ( if_inpoly(rlong ,rlat,polygon%polygons(i)%coords).ne.0 &
140  .or.if_inpoly(rlong2 ,rlat,polygon%polygons(i)%coords).ne.0 ) then
141  iok=1
142  return
143  endif
144  endif
145  enddo
146  ! not inside any polygon%polygons; set iok to 0 if there are any we should have
147  ! been in
148  iok = 1
149  if(ianyok.gt.0) iok = 0
150 
151  return
152 end subroutine

154 ! ==============================================================================
155 !! taken from spotl \cite Agnew97
156 !! \par oryginal comment:
157 !! Rewritten by D. Agnew from the version by Godkin and Pulli,
158 !! in BSSA, Vol 74, pp 1847-1848 (1984)
159 !! adopted and slightly modified M. Rajner
160 !! cords is x, y (lon, lat) 2 dimensional array
161 ! ==============================================================================
162 integer function if_inpoly(x,y,coords)
163  use mod_constants, only: dp, dp
164  real(dp) ,allocatable , dimension (:,:) , intent (in) :: coords
165  real(dp) , intent (in) :: x , y
166  integer :: i , isc
167  ! Returns 1 if point at (x,y) is inside polygon whose nv vertices
168  ! Returns 0 if point is outside
169  ! Returns 2 if point is on edge or vertex
170 
171  if_inpoly = 0
172  do i=1, size(coords(:,1))-1
173  isc = ncross( &
174  coords(i,1) - x, &
175  coords(i,2) - y, &
176  coords(i+1,1) - x, &
177  coords(i+1,2) - y )
178  ! on edge - know the answer
179  if(isc.eq.4) then
180  if_inpoly = 2
181  return
182  endif
183  if_inpoly = if_inpoly + isc
184  enddo
185 ! check final segment
186  isc = ncross( &
187  coords(size(coords(:,1)) , 1 ) - x , &
188  coords(size(coords(:,2)) , 2 ) - y , &
189  coords(1 , 1 ) - x , &
190  coords(1 , 2 ) - y )
191  if(isc.eq.4) then
192  if_inpoly = 2
193  return
194  endif
195  if_inpoly = if_inpoly + isc
196  if_inpoly = if_inpoly/2
197  ! convert to all positive (a departure from the original)
198  if_inpoly = iabs(if_inpoly)
199  return
200 end function

202 ! ==============================================================================
203 !> \brief finds whether the segment from point 1 to point 2 crosses
204 !! the negative x-axis or goes through the origin (this is
205 !! the signed crossing number)
206 !!
207 !! return value nature of crossing
208 !! 4 segment goes through the origin
209 !! 2 segment crosses from below
210 !! 1 segment ends on -x axis from below
211 !! or starts on it and goes up
212 !! 0 no crossing
213 !! -1 segment ends on -x axis from above
214 !! or starts on it and goes down
215 !! -2 segment crosses from above
216 !!
217 !! taken from spotl \cite Agnew97
218 !! slightly modified
219 ! ==============================================================================
220 integer function ncross(x1,y1,x2,y2)
221  use mod_constants, only: dp, dp
222  use mod_cmdline
223  real(dp) , intent(in) :: x1 , y1, x2 , y2
224  real(dp) :: c12 , c21
225 
226  ! all above (or below) axis
227  if(y1*y2.gt.0) then
228  ncross = 0
229  return
230  endif
231 
232  c12 = x1*y2
233  c21 = x2*y1
234 
235  ! through origin
236  if(c12.eq.c21.and.x1*x2.le.0.) then
237  ncross = 4
238  return
239  endif
240 
241  ! touches +x axis; crosses +x axis; lies entirely on -x axis
242  if( (y1.eq.0.and.x1.gt.0) &
243  .or.(y2.eq.0.and.x2.gt.0) &
244  .or.((y1.lt.0).and.(c12.gt.c21)) &
245  .or.((y1.gt.0).and.(c12.lt.c21)) &
246  .or.(y1.eq.0.and.y2.eq.0.and.x1.lt.0.and.x2.lt.0)) &
247  then
248  ncross = 0
249  return
250  endif
251 
252  ! cross axis
253  if(y1.ne.0.and.y2.ne.0) then
254  if(y1.lt.0) ncross = 2
255  if(y1.gt.0) ncross = -2
256  return
257  endif
258 ! one end touches -x axis - goes which way?
259  if(y1.eq.0) then
260  if(y2.lt.0) ncross = -1
261  if(y2.gt.0) ncross = 1
262  else
263 ! y2=0 - ends on x-axis
264  if(y1.lt.0) ncross = 1
265  if(y1.gt.0) ncross = -1
266  endif
267  return
268 end function

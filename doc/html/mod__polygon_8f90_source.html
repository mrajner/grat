143 ! ==============================================================================
144 !> Check if point is in closed polygon
145 !!
146 !! From spotl \cite Agnew97
147 !! adopted to \c grat and Fortran90 syntax
148 !! From original description
149 !! returns iok=0 if
150 !! 1. there is any polygon (of all those read in) in which the
151 !! coordinate should not fall, and it does
152 !! or
153 !! 2. the coordinate should fall in at least one polygon
154 !! (of those read in) and it does not
155 !! otherwise returns iok=1
156 !! \author D.C. Agnew \cite Agnew96
157 !! \author adopted by Marcin Rajner
158 !! \date 2013-03-04
159 !!
160 !! The ilustration explain exclusion idea\n
161 !! \image latex /home/mrajner/src/grat/doc/figures/polygon_ilustration.pdf "capt" width=\textwidth
162 !! \image html /home/mrajner/src/grat/doc/figures/polygon_ilustration.png
163 ! ==============================================================================
164 subroutine chkgon (rlong , rlat , polygon , iok)
165  real(dp),intent (in) :: rlong, rlat
166  integer :: i, ianyok
167  integer(2) , intent (out) :: iok
168  real(dp) :: rlong2
169  type(polygon_info) , intent (in) :: polygon
170 
171  ! ! Check first if we need to use this soubroutine
172  if (size(polygon%polygon).eq.0) then
173  iok=0
174  return
175  endif
176 
177  if(rlong.gt.180) rlong2 = rlong - 360.
178  ! loop over polygons
179  do i=1,size(polygon%polygon)
180  ! loop twice for elastic and newtonian
181  ! polygon is one we should not be in
182  if(.not.polygon%polygon(i)%use) then
183  if ( if_inpoly(rlong ,rlat,polygon%polygon(i)%coords).ne.0 &
184  .or.if_inpoly(rlong2 ,rlat,polygon%polygon(i)%coords).ne.0 ) then
185  iok=0
186  return
187  endif
188  endif
189 enddo
190 ianyok=0
191 ! polygon is one we should be in; test to see if we are, and if so set
192 ! iok to 1 and return
193 do i=1,size(polygon%polygon)
194  if(polygon%polygon(i)%use) then
195  ianyok = ianyok+1
196  if ( if_inpoly(rlong ,rlat,polygon%polygon(i)%coords).ne.0 &
197  .or.if_inpoly(rlong2 ,rlat,polygon%polygon(i)%coords).ne.0 ) then
198  iok=1
199  return
200  endif
201 endif
202  enddo
203  ! not inside any polygon%polygons; set iok to 0 if there are any we should have
204  ! been in
205  iok = 1
206  if(ianyok.gt.0) iok = 0
207  return
208 end subroutine

210 ! ==============================================================================
211 !! taken from spotl \cite Agnew97
212 !! \par oryginal comment:
213 !! Rewritten by D. Agnew from the version by Godkin and Pulli,
214 !! in BSSA, Vol 74, pp 1847-1848 (1984)
215 !! adopted and slightly modified M. Rajner
216 !! cords is x, y (lon, lat) 2 dimensional array
217 ! ==============================================================================
218 integer function if_inpoly(x,y,coords)
219  use mod_constants, only: dp, dp
220  real(dp) ,allocatable , dimension (:,:) , intent (in) :: coords
221  real(dp) , intent (in) :: x , y
222  integer :: i , isc
223  ! Returns 1 if point at (x,y) is inside polygon whose nv vertices
224  ! Returns 0 if point is outside
225  ! Returns 2 if point is on edge or vertex
226 
227  if_inpoly = 0
228  do i=1, size(coords(:,1))-1
229  isc = ncross( &
230  coords(i,1) - x, &
231  coords(i,2) - y, &
232  coords(i+1,1) - x, &
233  coords(i+1,2) - y )
234  ! on edge - know the answer
235  if(isc.eq.4) then
236  if_inpoly = 2
237  return
238  endif
239  if_inpoly = if_inpoly + isc
240  enddo
241  ! check final segment
242  isc = ncross( &
243  coords(size(coords(:,1)) , 1 ) - x , &
244  coords(size(coords(:,2)) , 2 ) - y , &
245  coords(1 , 1 ) - x , &
246  coords(1 , 2 ) - y )
247  if(isc.eq.4) then
248  if_inpoly = 2
249  return
250  endif
251  if_inpoly = if_inpoly + isc
252  if_inpoly = if_inpoly/2
253  ! convert to all positive (a departure from the original)
254  if_inpoly = iabs(if_inpoly)
255  return
256 end function

258 ! ==============================================================================
259 !> \brief finds whether the segment from point 1 to point 2 crosses
260 !! the negative x-axis or goes through the origin (this is
261 !! the signed crossing number)
262 !!
263 !! return value nature of crossing
264 !! 4 segment goes through the origin
265 !! 2 segment crosses from below
266 !! 1 segment ends on -x axis from below
267 !! or starts on it and goes up
268 !! 0 no crossing
269 !! -1 segment ends on -x axis from above
270 !! or starts on it and goes down
271 !! -2 segment crosses from above
272 !!
273 !! taken from spotl \cite Agnew97
274 !! slightly modified
275 ! ==============================================================================
276 integer function ncross(x1,y1,x2,y2)
277  real(dp) , intent(in) :: x1 , y1, x2 , y2
278  real(dp) :: c12 , c21
279 
280  ! all above (or below) axis
281  if(y1*y2.gt.0) then
282  ncross = 0
283  return
284  endif
285 
286  c12 = x1*y2
287  c21 = x2*y1
288 
289  ! through origin
290  if(c12.eq.c21.and.x1*x2.le.0.) then
291  ncross = 4
292  return
293  endif
294 
295  ! touches +x axis; crosses +x axis; lies entirely on -x axis
296  if( (y1.eq.0.and.x1.gt.0) &
297  .or.(y2.eq.0.and.x2.gt.0) &
298  .or.((y1.lt.0).and.(c12.gt.c21)) &
299  .or.((y1.gt.0).and.(c12.lt.c21)) &
300  .or.(y1.eq.0.and.y2.eq.0.and.x1.lt.0.and.x2.lt.0)) &
301  then
302  ncross = 0
303  return
304  endif
305 
306  ! cross axis
307  if(y1.ne.0.and.y2.ne.0) then
308  if(y1.lt.0) ncross = 2
309  if(y1.gt.0) ncross = -2
310  return
311  endif
312  ! one end touches -x axis - goes which way?
313  if(y1.eq.0) then
314  if(y2.lt.0) ncross = -1
315  if(y2.gt.0) ncross = 1
316  else
317  ! y2=0 - ends on x-axis
318  if(y1.lt.0) ncross = 1
319  if(y1.gt.0) ncross = -1
320  endif
321  return
322 end function

141 subroutine progress(j, time, cpu, every)
142  use mod_constants, only: dp
143  use mod_cmdline, only: moreverbose, quiet, quiet_step
144  use iso_fortran_env, only: output_unit
145 
146  implicit none
147  integer(kind=4)::j, k
148  integer:: ii
149  character(len=27)::bar="???% | |"
150  real :: time, cpu
151  integer, optional :: every
152  integer :: every_
153  integer,save :: step=0
154  character(len=1) :: timeunit
155  logical :: logprinted
156 
157 
158  if (present(every)) then
159  every_=every
160  else
161  every_=quiet_step
162  endif
163 
164  step = step + 1
165 
166  if ((every_.eq.0).and.j.ne.100) then
167  return
168  else if (every_.eq.0.and.j.eq.100) then
169  else if ( &
170  modulo(step,every_).ne.0 &
171  .and.j.ne.every_ &
172  .and.j.ne.100 &
173  .and.step.ne.1 &
174  ) then
175  return
176  endif
177 
178  write(unit=bar(1:3),fmt="(i3)") j
179  do k=1, j/5
180  bar(6+k:6+k)="*"
181  enddo
182 
183  if (time.gt.60000) then
184  time = time/3600
185  cpu = cpu/3600
186  timeunit = "h"
187  else if (time.gt.1000) then
188  time = time/60
189  cpu = cpu/60
190  timeunit = "m"
191  else
192  timeunit="s"
193  endif
194 
195 
196  if (.not.(quiet.or.output%unit.eq.output_unit)) then
197  open (unit=output_unit, carriagecontrol='fortran')
198  write( &
199  unit=output_unit, &
200  fmt="(a1,a1,a27, &
201  f5.1,a1,1x,a,f5.1,a,1x, &
202  a,f5.1,x, &
203  a,f5.1,a1, &
204  x,a,<size(moreverbose)+1>(x,a)$)" &
205  ) &
206  '+',char(13), bar, &
207  time, timeunit, "[eta", 100.*time/j,"]", &
208  "(proc:",cpu, &
209  "| %:",cpu/time*100,")", &
210  trim(output%name), &
211  ( &
212  trim(moreverbose(ii)%name), ii=1, size(moreverbose) &
213  )
214  endif
215 
216  if (j.eq.100.and..not.logprinted) then
217  close(output_unit)
218  write(log%unit, &
219  '("Execution time:",1x,f5.1,a, &
220  &" (proc time:",1x,f5.1,1x,"|%", f5.1,")")') &
221  time, timeunit, &
222  cpu, &
223  cpu/time*100
224  logprinted=.true.
225  endif
226  return
227 end subroutine progress

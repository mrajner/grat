5 ! =============================================================================
6 !> Calculate area of spherical segment
7 !!
8 !! Computes spherical area on unit (default if optional argument \c radius is
9 !! not given) sphere given by:
10 !! - method 1 (\c alternative_method not given or \c alternative_method .false.)
11 !! - distance from station, segment size in spher distance and angle
12 !! - method 2 (\c alternative_method .true.)
13 !! - distance from station start, distance from station end
14 !!
15 !! The ilustration explain optional \c method argument
16 !! \latexonly
17 !! \begin{center}
18 !! \tikzsetfigurename{spher_area}
19 !! \input{/home/mrajner/src/grat/doc/figures/spher_area}
20 !! \end{center}
21 !! \endlatexonly
22 !! \image html /home/mrajner/src/grat/doc/figures/spher_area.svg
23 !!
24 !! \warning All input angles in radians, output area on unit sphere or
25 !! in square units of given (optionally) \c radius.
26 ! =============================================================================
27 function spher_area (distance, ddistance, azstp, radius, alternative_method)
28  real(dp) :: spher_area
29  real(dp), intent(in) :: distance, ddistance
30  real(dp), intent(in) :: azstp
31  logical, intent(in), optional :: alternative_method
32  real(dp), intent(in), optional :: radius
33 
34  if (present(alternative_method).and.alternative_method) then
35  spher_area = (-cos(ddistance) + cos(distance))*(azstp)
36  else
37  spher_area = (-cos(distance+ddistance/2.) + cos(distance-ddistance/2.))*(azstp)
38  endif
39 
40  if(present(radius)) spher_area = spher_area * radius**2
41 
42 end function

44 ! =============================================================================
45 !> This soubroutine gives the latitude and longitude of the point at the
46 !! specified distance and azimuth from site latitude and longitude.
47 !!
48 !! all parameters in decimal degree
49 !! \author D.C. Agnew \cite Agnew96
50 !! \date 2012
51 !! \author M. Rajner - modification and adpaptation
52 !! \date 2013-03-06
53 !! \warning all values in radians
54 ! =============================================================================
55 subroutine spher_trig (latin, lonin, distance, azimuth, latout, lonout, domain)
56  real(dp), intent(in) :: distance
57  real(dp), intent(in) :: latin, lonin, azimuth
58  real(dp), intent(out) :: latout, lonout
59  real(dp) :: saz, caz, st, ct, cd, sd, cb, sb
60  logical, intent(in), optional :: domain
61 
62  ct = cos(pi/2.-latin)
63  st = sin(pi/2.-latin)
64  cd = cos(distance)
65  sd = sin(distance)
66  saz = sin(azimuth)
67  caz = cos(azimuth)
68  cb = cd*ct + sd*st*caz
69  !if(abs(cb).gt.1) cb = cb/abs(cb)
70  sb = sqrt(1.-cb**2)
71  latout = pi/2. - acos(cb)
72  lonout = lonin + atan2(sd*saz/sb,(st*cd - sd*ct*caz)/sb)
73 
74  if (present(domain).and.domain) then
75  if (lonout.lt.0) then
76  lonout=lonout+2*pi
77  endif
78  endif
79 end subroutine

81 ! =============================================================================
82 !> For given coordinates for two points on sphere calculate distance and
83 !! azimuth in radians
84 !!
85 !! Input coordinates ub
86 !! \author M. Rajner
87 !! \date 2013-03-04
88 !! for small spherical distances you should always use havesine=.true.
89 !!
90 !! All arguments in radians
91 ! =============================================================================
92 subroutine spher_trig_inverse (lat1, lon1, lat2, lon2, distance, azimuth, haversine)
93  real(dp), intent (in) :: lat1, lon1, lat2, lon2
94  real(dp), intent (out) :: distance, azimuth
95  real(dp) :: dlat, dlon, a
96  logical, intent(in), optional :: haversine
97 
98  dlon = lon2 - lon1
99  dlat = lat2 - lat1
100 
101  if (dlon > pi) dlon=dlon-pi
102 
103  if (present(haversine).and.haversine) then
104  ! the formula below from Robert Chamberlain
105  ! http://www.usenet-replayer.com/faq/comp.infosystems.gis.html
106  a = (sin(dlat/2))**2 + cos(lat1) * cos(lat2) * (sin(dlon/2))**2
107  distance = 2 * atan2( sqrt(a), sqrt(1-a) )
108  else
109  distance = acos( sin(lat1) * sin(lat2) + cos(lat1) * cos(lat2) * cos(dlon))
110  endif
111 
112  azimuth = &
113  atan2( &
114  (sin(dlon)*cos(lat2)/sin(distance)), &
115  ((sin(lat2)*cos(lat1) - cos(lat2)*sin(lat1)*cos(dlon))/sin(distance)) &
116  )
117 
118  if (azimuth.lt.0) azimuth = azimuth + 2 * pi
119 end subroutine

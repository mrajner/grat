33 ! ==============================================================================
34 !> Compute coefficients for spline interpolation.
35 !!
36 !! From web sources \todo find url
37 !! Original description below:
38 !! ==============================================================================
39 !! Calculate the coefficients b(i), c(i), and d(i), i=1,2,...,n
40 !! for cubic spline interpolation
41 !! s(x) = y(i) + b(i)*(x-x(i)) + c(i)*(x-x(i))**2 + d(i)*(x-x(i))**3
42 !! for x(i) <= x <= x(i+1)
43 !! Alex G: January 2010
44 !!
45 !! input..
46 !! x = the arrays of data abscissas (in strictly increasing order)
47 !! y = the arrays of data ordinates
48 !! n = size of the arrays xi() and yi() (n>=2)
49 !! output..
50 !! b, c, d = arrays of spline coefficients
51 !! comments ...
52 !! spline.f90 program is based on fortran version of program spline.f
53 !! the accompanying function fspline can be used for interpolation
54 !! ==============================================================================
55 subroutine spline (x, y, b, c, d, n)
56  integer n
57  real(dp) :: x(n), y(n), b(n), c(n), d(n)
58  integer i, j, gap
59  real(dp) :: h
60 
61  gap = n-1
62  ! check input
63  if ( n < 2 ) return
64  if ( n < 3 ) then
65  b(1) = (y(2)-y(1))/(x(2)-x(1)) ! linear interpolation
66  c(1) = 0.
67  d(1) = 0.
68  b(2) = b(1)
69  c(2) = 0.
70  d(2) = 0.
71  return
72  end if
73  !
74  ! step 1: preparation
75  !
76  d(1) = x(2) - x(1)
77  c(2) = (y(2) - y(1))/d(1)
78  do i = 2, gap
79  d(i) = x(i+1) - x(i)
80  b(i) = 2.0*(d(i-1) + d(i))
81  c(i+1) = (y(i+1) - y(i))/d(i)
82  c(i) = c(i+1) - c(i)
83  end do
84  !
85  ! step 2: end conditions
86  !
87  b(1) = -d(1)
88  b(n) = -d(n-1)
89  c(1) = 0.0
90  c(n) = 0.0
91  if(n /= 3) then
92  c(1) = c(3)/(x(4)-x(2)) - c(2)/(x(3)-x(1))
93  c(n) = c(n-1)/(x(n)-x(n-2)) - c(n-2)/(x(n-1)-x(n-3))
94  c(1) = c(1)*d(1)**2/(x(4)-x(1))
95  c(n) = -c(n)*d(n-1)**2/(x(n)-x(n-3))
96  end if
97  !
98  ! step 3: forward elimination
99  !
100  do i = 2, n
101  h = d(i-1)/b(i-1)
102  b(i) = b(i) - h*d(i-1)
103  c(i) = c(i) - h*c(i-1)
104  end do
105  !
106  ! step 4: back substitution
107  !
108  c(n) = c(n)/b(n)
109  do j = 1, gap
110  i = n-j
111  c(i) = (c(i) - d(i)*c(i+1))/b(i)
112  end do
113  !
114  ! step 5: compute spline coefficients
115  !
116  b(n) = (y(n) - y(gap))/d(gap) + d(gap)*(c(gap) + 2.0*c(n))
117  do i = 1, gap
118  b(i) = (y(i+1) - y(i))/d(i) - d(i)*(c(i+1) + 2.0*c(i))
119  d(i) = (c(i+1) - c(i))/d(i)
120  c(i) = 3.*c(i)
121  end do
122  c(n) = 3.0*c(n)
123  d(n) = d(n-1)
124 end subroutine spline

126 ! ==============================================================================
127 !> Evaluates the cubic spline interpolatione.
128 !!
129 ! ==============================================================================
130 !> Function ispline evaluates the cubic spline interpolation at point z
131 !! ispline = y(i)+b(i)*(u-x(i))+c(i)*(u-x(i))**2+d(i)*(u-x(i))**3
132 !! where x(i) <= u <= x(i+1)
133 !!----------------------------------------------------------------------
134 !! input..
135 !! u = the abscissa at which the spline is to be evaluated
136 !! x, y = the arrays of given data points
137 !! b, c, d = arrays of spline coefficients computed by spline
138 !! n = the number of data points
139 !! output:
140 !! ispline = interpolated value at point u
141 !!
142 !! \date 2013-03-10
143 !! \author M. Rajner
144 !! added optional parameter method
145 !!=======================================================================
146 real (dp) function ispline(u, x, y, b, c, d, n, method)
147  integer n
148  real(dp):: u, x(n), y(n), b(n), c(n), d(n)
149  integer :: i, j, k
150  real(dp) :: dx
151  character(*), optional :: method
152 
153  ! if u is ouside the x() interval take a boundary value (left or right)
154  if(u <= x(1)) then
155  ispline = y(1)
156  return
157  end if
158  if(u >= x(n)) then
159  ispline = y(n)
160  return
161  end if
162 
163  !*
164  ! binary search for for i, such that x(i) <= u <= x(i+1)
165  !*
166  i = 1
167  j = n+1
168  do while (j > i+1)
169  k = (i+j)/2
170  if(u < x(k)) then
171  j=k
172  else
173  i=k
174  end if
175  end do
176  !*
177  ! evaluate spline interpolation
178  !*
179  dx = u - x(i)
180 
181  if (present(method)) then
182  if (method == "nearest") then
183  if ((x(i+1)-u) < dx) then
184  ispline = y(i+1)
185  return
186  else
187  ispline = y(i)
188  return
189  endif
190  elseif (method == "linear") then
191  ispline = y(i) + (y(i+1)-y(i))/(x(i+1)-x(i)) * dx
192  return
193  endif
194  endif
195  ispline = y(i) + dx*(b(i) + dx*(c(i) + dx*d(i)))
196 end function ispline

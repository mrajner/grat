\documentclass[beamer]{standalone}
\usepackage{tikz}

\tikzset{
  small dot/.style={
    inner sep=0,
    minimum size=1mm,
    fill,
    circle,
  }
}

\begin{document}
  \newcounter{komorka}\setcounter{komorka}{0}
  \begin{standaloneframe}
    \only<1>{\def\cell{13}}
    \only<2>{\setcounter{komorka}{0}\def\cell{23}}
  \begin{tikzpicture}[scale=0.3]
    \draw[gray,step=5] (-12.4,-20) grid (17,15);

    \coordinate(start) at (2,-0.5);
    \foreach \startradius/\endradius/\ileaz in {
      0/1/4,
      1/4/6,
      4/8/10,
      8/14/12
    }{
      \draw(start) circle (\endradius);
      \pgfmathsetmacro{\da}{360/\ileaz}
      \foreach \az in {
        1,...,\ileaz
      }{
        \pgfmathsetmacro{\a}{\da/2+\az*\da}
        \draw (start)++(\a:\startradius) -- ++(\a:\endradius-\startradius);

        \addtocounter{komorka}{1}
        \ifnum\thekomorka=\cell
        \draw[fill=green!\thekomorka] (start)++(\a:\startradius)
        --+ (\a:\endradius-\startradius)
        arc (\a:\a+\da:\endradius)
        --+ (\a+\da:\startradius-\endradius)
        arc (\a+\da:\a:\startradius);

        \path (start) ++(\a+\da/2:\startradius)
        --+(\a+\da/2:\endradius/2-\startradius/2)
        node[small dot]{} 
        node(centroid-\thekomorka){\thekomorka}
        ;
        \fi

      }
    }
  \end{tikzpicture}%
\end{standaloneframe}

\end{document}
